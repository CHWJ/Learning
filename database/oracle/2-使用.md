### 直接重建序列法
``` sql
select SEQ_NODE_INFO.nextval from dual;
select MAX(ID) - 3147 from T_NODE_INFO; -- 57
ALTER SEQUENCE SEQ_NODE_INFO INCREMENT BY 57;
select SEQ_NODE_INFO.nextval from dual;
ALTER SEQUENCE SEQ_NODE_INFO INCREMENT BY 1;
```

### 创建用户
``` sql
CREATE USER c##my_schema IDENTIFIED BY my_password DEFAULT TABLESPACE NEW_SCHEMA_DATA TEMPORARY TABLESPACE NEW_SCHEMA_TEMP;
```

### 授权
``` sql
-- 授予最基础的连接和开发角色
GRANT CONNECT, RESOURCE TO c##my_schema;

-- 授予在表空间上的无限额度（否则无法创建表并插入数据）
GRANT UNLIMITED TABLESPACE TO c##my_schema;
```

### 备份数据库
``` sql

-- 以 sysdba 身份登录
sqlplus / as sysdba

-- 1. 创建物理目录（确保操作系统中该路径真实存在，且 oracle 用户有读写权限）
CREATE OR REPLACE DIRECTORY dpump_dir AS '/data/backup'
chown -R oracle:oinstall /data/backup
chmod -R 775 /data/backup

-- 2. 将该目录的读写权限赋给需要执行导出的用户（假设你要导出的模式是 scott）
GRANT READ, WRITE ON DIRECTORY dpump_dir TO scott

-- Oracle 11g 特别避坑指南：空表无法导出的问题
-- 查询所有没有分配段的空表，并生成修改语句，然后执行
SELECT 'ALTER TABLE ' || table_name || ' ALLOCATE EXTENT;' FROM user_tables WHERE segment_created = 'NO';

-- 查看当前大小（如果为 0 且开启了 ASMM，则由系统动态分配，但有时反应较慢）
SHOW PARAMETER streams_pool_size;

-- 临时调大该参数（立刻生效，不需要重启数据库）
ALTER SYSTEM SET streams_pool_size=128M SCOPE=BOTH;

-- 在操作系统命令行（注意：不是 SQLPlus 内）执行
expdp scott/password@orcl directory=dpump_dir dumpfile=scott_schema_%U.dmp logfile=scott_exp.log schemas=scott  version=11.2

```

### 还原数据库

``` sql

sqlplus / as sysdba

-- 创建目录映射（注意：Windows 路径要用反斜杠，且路径两边加单引号）
CREATE OR REPLACE DIRECTORY win_dpump_dir AS 'D:\backup\dpdump';

-- 把读写权限赋给你要导入的用户，或者直接赋给 system
GRANT READ, WRITE ON DIRECTORY win_dpump_dir TO system;

-- 创建数据表空间
CREATE TABLESPACE YCYT_DATA DATAFILE 'D:\oracle_11g\oradata\orcl\ycyt_data01.dbf' SIZE 2G AUTOEXTEND ON NEXT 100M MAXSIZE UNLIMITED;

-- 2. 创建临时表空间
CREATE TEMPORARY TABLESPACE YCYT_TEMP TEMPFILE 'D:\oracle_11g\oradata\orcl\ycyt_temp01.dbf' SIZE 1G AUTOEXTEND ON NEXT 50M MAXSIZE UNLIMITED;

-- 3. 用户并关联表空间
    -- 将已存在的用户 YCYTDBA 关联到指定的表空间上
    ALTER USER ycytdba DEFAULT TABLESPACE YCYT_DATA;
    -- 将已存在的用户 YCYTDBA 关联到指定的临时表空间上
    ALTER USER ycytdba TEMPORARY TABLESPACE YCYT_TEMP;

-- 4. 赋予基础和数据泵权限
GRANT CONNECT, RESOURCE TO ycytdba;
GRANT DATAPUMP_IMP_FULL_DATABASE TO ycytdba;
-- 如果涉及一些特殊的系统权限，可以先给个 dba（可选）
GRANT DBA TO ycytdba;

-- 查看当前大小（如果为 0 且开启了 ASMM，则由系统动态分配，但有时反应较慢）
SHOW PARAMETER streams_pool_size;

-- 在 11g 中，如果 streams_pool_size 初始值为 0，当 impdp 突然需要大量并发队列内存时，Oracle 的自动调整（ASMM）往往反应太慢，导致导入命令在刚启动或者换轨时严重卡顿
ALTER SYSTEM SET STREAMS_POOL_SIZE = 256M SCOPE=MEMORY;

impdp ycytdba/你的system密码@新数据库实例名 directory=win_dpump_dir dumpfile=ycytdba_schema_bak_%U.dmp logfile=ycyt_imp.log schemas=ycytdba remap_tablespace=原表空间:YCYT_DATA table_exists_action=REPLACE PARALLEL=4
```

### 追加数据文件

``` sql
ALTER TABLESPACE YCYT_DATA 
ADD DATAFILE 'D:\app\oracle\oradata\orcl\ycyt_data02.dbf' SIZE 2G AUTOEXTEND ON NEXT 100M MAXSIZE UNLIMITED;

```

### 设置用户密码

``` sql
-- 在 Oracle 中，标准的密码是不区分大小写的（默认会全部转为大写）
ALTER USER 用户名 IDENTIFIED BY 新密码;
-- 如果密码中必须包含 - 等特殊字符，你需要使用双引号 "" 将密码包裹起来。
ALTER USER 用户名 IDENTIFIED BY "my-new-password";
```

### 解锁用户

``` sql
-- 解锁因输错密码被锁定的用户
ALTER USER 用户名 ACCOUNT UNLOCK;
```

### 检查Oracle 监听器状态

``` bash
# 切换到 oracle 用户
su - oracle
lsnrctl status
```

### 启动Oracle 监听器

``` bash
su - oracle
lsnrctl start
```
